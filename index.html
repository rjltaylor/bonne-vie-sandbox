    let activeData = JSON.parse(localStorage.getItem('bonneVieSandboxData')) || initialData;


    function renderTable() {

        const tbody = document.getElementById('sandbox-tbody');

        tbody.innerHTML = '';


        activeData.forEach((row, index) => {

            const tr = document.createElement('tr');

    function updateValue(index, field, value) {

        activeData[index][field] = value;

        localStorage.setItem('bonneVieSandboxData', JSON.stringify(activeData));

        calculateMetrics(); // Real-time calculation shifts

    }


    function addNewRow() {

        activeData.push({ day: `New Day ${activeData.length + 1}`, pillar: pillars[0], platform: platforms[0], caption: "", visual: "", status: statuses[0], notes: "" });

        localStorage.setItem('bonneVieSandboxData', JSON.stringify(activeData));

        renderTable();

    }


    function deleteRow(index) {

        if(confirm("Are you sure you want to remove this scheduling row?")) {

            activeData.splice(index, 1);

            localStorage.setItem('bonneVieSandboxData', JSON.stringify(activeData));

            renderTable();

        }

    }


    // Advanced dynamic metrics calculator

    function calculateMetrics() {

        const totalRows = activeData.length;

        if(totalRows === 0) return;


        const promoCount = activeData.filter(r => r.pillar === "Book Promotion 📖").length;

        const promoPercentage = Math.round((promoCount / totalRows) * 100);

        

        document.getElementById('balance-stat').innerText = `${promoPercentage}%`;

        document.getElementById('balance-bar').style.width = `${promoPercentage}%`;

        

        const badge = document.getElementById('balance-badge');

        if (promoPercentage <= 25) {

            badge.innerText = "Balanced ✅";

            badge.style.backgroundColor = "var(--sage-light)";

        } else {

            badge.innerText = "Heavy Promo ⚠️";

            badge.style.backgroundColor = "#FDF2F2";

        }


        const drafts = activeData.filter(r => r.status === "☐ Draft").length;

        const scheduled = activeData.filter(r => r.status === "🔄 Scheduled").length;

        const live = activeData.filter(r => r.status === "✅ Live").length;


        document.getElementById('completion-stat').innerText = `${scheduled + live} / ${totalRows}`;

        document.getElementById('status-breakdown').innerText = `Drafts: ${drafts} | Scheduled: ${scheduled} | Live: ${live}`;


        // Find Top Pillar Focus

        const counts = {};

        activeData.forEach(r => counts[r.pillar] = (counts[r.pillar] || 0) + 1);

        const topPillar = Object.keys(counts).reduce((a, b) => counts[a] > counts[b] ? a : b, pillars[0]);

        document.getElementById('top-pillar').innerText = topPillar;

    }


    // Collaboration Save File Generator

    function downloadBackup() {

        const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(activeData, null, 2));

        const downloadAnchor = document.createElement('a');

        downloadAnchor.setAttribute("href", dataStr);

        downloadAnchor.setAttribute("download", "georgie_provence_workspace.json");

        document.body.appendChild(downloadAnchor);

        downloadAnchor.click();

        downloadAnchor.remove();

    }


    // Collaboration Load File Processor

    function uploadBackup(event) {

        const file = event.target.files[0];

        if (!file) return;

        const reader = new FileReader();

        reader.onload = function(e) {

            try {

                const parsed = JSON.parse(e.target.result);

                if (Array.isArray(parsed)) {

                    activeData = parsed;

                    localStorage.setItem('bonneVieSandboxData', JSON.stringify(activeData));

                    renderTable();

                    alert("Workspace updated successfully with the loaded files!");

                }

            } catch (err) {

                alert("Error reading file configuration. Ensure it's a valid dashboard backup.");

            }

        };

        reader.readAsText(file);

    }


    // CSV Spreadsheet Exporter

    function exportToCSV() {

        let csvContent = "data:text/csv;charset=utf-8,Day/Date,Core Pillar,Platform,Caption,Visual Asset,Status,Coaching Notes\n";

        activeData.forEach(r => {

            csvContent += rowStr + "\n";

        });

        const encodedUri = encodeURI(csvContent);

        const link = document.createElement("a");

        link.setAttribute("href", encodedUri);

        link.setAttribute("download", "bonne_vie_content_export.csv");

        document.body.appendChild(link);

        link.click();

        link.remove();

    }
